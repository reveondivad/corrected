const express = require('express') const router = express.Router() const escape = require('escape-html'); router.get('/greeting', (req, res) => { const { name } = req.query; res.send(`

Hello : ${escape(name)}

`); }) router.get('/greet-template', (req,res) => { const name = escape(req.query.name); res.render('index', { user_name: name }); }) module.exports = router
